214 void MainWindow::showData(Core::PtrFilter filter)
215 {
216  QColor color = m_colorManager.nextColor();
217  QString fname = QString::fromStdString(filter->info()->name());
218 
219  QPen mxPen, mePen, sxPen, sePen;
220  mxPen.setWidthF(2.0);
221  mxPen.setColor(Qt::darkMagenta);
222  mePen.setWidthF(2.0);
223  mePen.setColor(color);
224  mePen.setStyle(Qt::DashDotLine);
225  sxPen.setWidthF(1.5);
226  sxPen.setColor(Qt::darkGray);
227  sxPen.setStyle(Qt::DashLine);
228  sePen.setWidthF(1.5);
229  sePen.setColor(color);
230 
231  int dim = int(filter->result()[0].meanX.size());
232  if (m_graphWindow->countSheets() != dim) {
233  m_graphWindow->setCountSheets(dim);
234  }
235 
236  QString title = tr("Статистика <") + m_taskWidget->name() + QString(">");
237  QString subTitle =
238  tr("Размер выборки ") + QString::number(m_filterParamsWidget->parameters()->sampleSize()) +
239  tr(", шаг интегрирования ") + QString::number(m_filterParamsWidget->parameters()->integrationStep()) +
240  tr(", между измерениями ") + QString::number(m_filterParamsWidget->parameters()->measurementStep());
241  for (int i = 0; i < dim; i++) {
242  m_graphWindow->sheet(i).setTitleLabel(title);
243  m_graphWindow->sheet(i).setSubTitleLabel(subTitle);
244  }
245  if (m_taskWidget->id() == Tasks::TASK_ID::Landing) {
246  m_graphWindow->sheet(0).setXLabel(tr("Время (с)"));
247  m_graphWindow->sheet(1).setXLabel(tr("Время (с)"));
248  m_graphWindow->sheet(2).setXLabel(tr("Время (с)"));
249  m_graphWindow->sheet(0).setYLabel(tr("Скорость (м/c)"));
250  m_graphWindow->sheet(1).setYLabel(tr("Угол наклона (°)"));
251  m_graphWindow->sheet(2).setYLabel(tr("Высота (м)"));
252  }
253 
254  Math::Vector scale(dim);
255  if (m_taskWidget->id() == Tasks::TASK_ID::Landing) {
256  scale[0] = scale[2] = 1000.0;
257  scale[1] = Math::Convert::RadToDeg(1.0);
258  } else {
259  for (int i = 0; i < dim; ++i) {
260  scale[i] = 1.0;
261  }
262  }
263 
264  QVector<double> x, y;
265  Core::GetTime(filter->result(), x);
266 
267  for (int i = 0; i < dim; i++) {
268  Core::GetMeanX(filter->result(), i, y, scale[i]);
269  m_graphWindow->sheet(i).addCurve(x, y, "Mx" + QString::number(i + 1), mxPen, false);
270 
271  Core::GetStdDeviationX(filter->result(), i, y, scale[i]);
272  m_graphWindow->sheet(i).addCurve(x, y, "Sx" + QString::number(i + 1), sxPen, false);
273 
274  Core::GetMeanE(filter->result(), i, y, scale[i]);
275  m_graphWindow->sheet(i).addCurve(x, y, "Me" + QString::number(i + 1) + " " + fname, mePen, false);
276 
277  Core::GetStdDeviationE(filter->result(), i, y, scale[i]);
278  m_graphWindow->sheet(i).addCurve(x, y, "Se" + QString::number(i + 1) + " " + fname, sePen, true);
279  }
280 
281  m_graphWindow->updatePlotter();
282 
283  addTable(filter->result(), filter->info()->name(), scale);
284 }
