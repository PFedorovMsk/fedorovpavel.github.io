131 void FilterResultsTable::writeToFile(QTextStream &out)
132 {
133  // найдем самые длинные строки в каждом столбце:
134  QVector<int> maxLength(m_table->columnCount());
135  for (int j = 0; j < m_table->columnCount(); ++j) {
136  maxLength[j] = 5;
137  }
138 
139  for (int i = 0; i < m_table->rowCount(); ++i) {
140  for (int j = 0; j < m_table->columnCount(); ++j) {
141  int length;
142  if ((length = m_table->item(i, j)->text().length()) > maxLength[j]) {
143  maxLength[j] = length;
144  }
145  }
146  }
147 
148  // сделаем линии для отчеркивания строк:
149  QString line = "|";
150  for (int j = 0; j < m_table->columnCount(); ++j) {
151  for (int k = 0; k < maxLength[j]; ++k) {
152  line += "-";
153  }
154  if (j != m_table->columnCount() - 1) {
155  line += "--+";
156  } else {
157  line += "--|\n";
158  }
159  }
160 
161  QString simpleLine = "";
162  for (int k = 0; k < line.length() - 1; ++k) {
163  simpleLine += "-";
164  }
165  simpleLine += "\n";
166 
167  QString titleLine = "";
168  titleLine += QString("| %1 |").arg("t", -maxLength[0]);
169  for (int j = 1; j < maxLength.size(); j += 4) {
170  int index = (j - 1) / 4 + 1;
171  titleLine += QString(" %1 |").arg("Mx" + QString::number(index), -maxLength[j]);
172  titleLine += QString(" %1 |").arg("Me" + QString::number(index), -maxLength[j + 1]);
173  titleLine += QString(" %1 |").arg("Sx" + QString::number(index), -maxLength[j + 2]);
174  titleLine += QString(" %1 |").arg("Se" + QString::number(index), -maxLength[j + 3]);
175  }
176  titleLine += "\n";
177 
178 
179  // запишем в поток:
180  QString str = "";
181  out << simpleLine << titleLine << line;
182  for (int i = 0; i < m_table->rowCount(); ++i) {
183  str = "|";
184  for (int j = 0; j < m_table->columnCount(); ++j) {
185  str += QString(" %1 |").arg(m_table->item(i, j)->text(), -maxLength[j]);
186  }
187  str += "\n";
188  out << str;
189  if (i < m_table->rowCount() - 1) {
190  out << line;
191  } else {
192  out << simpleLine;
193  }
194  }
