445 void GraphWindow::updatePlotter()
446 {

451  m_plotter->yAxis->setRange(range.yMin, range.yMax);

454  if (title.length() == 0) {
455  title = tr("Статистика <...>");
456  }
457  QString subTitle = m_currentSheet->subTitleLabel();
458  if (subTitle.length() == 0) {

463  plotTitle->setFont(m_plotterTitleFont);
464  m_plotter->plotLayout()->remove(m_plotter->plotLayout()->element(0, 0));
465  m_plotter->plotLayout()->addElement(0, 0, plotTitle);
466  m_plotter->plotLayout()->element(0, 0)->setAntialiased(true);
467 
468  QCPTextElement *plotSubTitle = new QCPTextElement(m_plotter, subTitle);
469  plotSubTitle->setFont(m_plotterSubTitleFont);
470  m_plotter->plotLayout()->remove(m_plotter->plotLayout()->element(1, 0));
471  m_plotter->plotLayout()->addElement(1, 0, plotSubTitle);
472  m_plotter->plotLayout()->element(1, 0)->setAntialiased(true);
473 
474  m_plotter->xAxis->setLabel(m_currentSheet->xLabel());
475  m_plotter->yAxis->setLabel(m_currentSheet->yLabel());
476 
477  for (int i = 0; i < m_currentSheet->curves().size(); i++) {
478  if (m_currentSheet->curves()[i].visible == false) {
479  continue;
480  }
481  int j = m_plotter->graphCount();

483  m_plotter->graph(j)->setData(m_currentSheet->curves()[i].x, m_currentSheet->curves()[i].y);
484  m_plotter->graph(j)->setPen(m_currentSheet->curves()[i].pen);

486  m_plotter->graph(j)->setAntialiased(true);
487  m_plotter->graph(j)->setAdaptiveSampling(true);
488  }
489 
490  m_plotter->replot();
491  updateMenu();
492 }
