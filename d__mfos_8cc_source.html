79 void MFOS::computeParams(size_t k, Array<Vector> &sampleU, Array<Vector> &sampleS, Matrix &T)
80 {
81  Vector my, chi;
82  Matrix Gamma, GammaY, GammaZ, DxyDxz, Ddelta, Dxy, Dxz;
83  Array<Vector> sampleDelta(m_params->sampleSize());
84 
85  // Индекс s пробегает по всем элементам выборки:
86  for (size_t s = 0; s < m_params->sampleSize(); ++s) {
87  MakeBlockVector(m_sampleY[s], m_sampleZ[s], sampleDelta[s]);
88  }
89 
90  Ddelta = Var(sampleDelta);
91  my = Mean(m_sampleY);
92  Dxy = Cov(m_sampleX, m_sampleY);
93  Dxz = Cov(m_sampleX, m_sampleZ);
94  MakeBlockMatrix(Dxy, Dxz, DxyDxz);
95 
96  Gamma = DxyDxz * Pinv(Ddelta);
97  GammaY = Gamma.leftCols(m_task->dimY());
98  GammaZ = Gamma.rightCols(m_task->dimX()); // dimZ = dimX
99 
100  chi = m_result[k].meanX - GammaY * my - GammaZ * m_result[k].meanZ;
101  T = m_result[k].varX - GammaY * Dxy.transpose() - GammaZ * Dxz.transpose();
102 
103  m_task->setTime(m_result[k].time);
104  // Индекс s пробегает по всем элементам выборки:
105  for (size_t s = 0; s < m_params->sampleSize(); ++s) {
106  sampleU[s] = GammaY * m_sampleY[s] + GammaZ * m_sampleZ[s] + chi;
107  sampleS[s] = m_task->tau(sampleU[s], T);
108  }
109 }

130
