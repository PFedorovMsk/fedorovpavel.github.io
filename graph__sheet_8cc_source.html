148 void GraphSheet::calcRanges()
149 {
150  if (!m_autoCalcRanges) {
151  m_axisRange = m_userAxisRange;
152  return;
153  }
154 
155  if (m_curves.size() == 0) {
156  m_axisRange.xMin = m_axisRange.yMin = 0;
157  m_axisRange.xMax = m_axisRange.yMax = 1;
158  return;
159  }
160 
161  double xmin = std::numeric_limits<double>::max();
162  double ymin = std::numeric_limits<double>::max();
163  double xmax = std::numeric_limits<double>::lowest();
164  double ymax = std::numeric_limits<double>::lowest();
165 
166  for (int i = 0; i < m_curves.size(); i++) {
167  if (m_curves[i].visible == false) {
168  continue;
169  }
170  for (int j = 0; j < m_curves[i].x.size(); j++) {
171  if (m_curves[i].x[j] > xmax) {
172  xmax = m_curves[i].x[j];
173  }
174  if (m_curves[i].y[j] > ymax) {
175  ymax = m_curves[i].y[j];
176  }
177  if (m_curves[i].x[j] < xmin) {
178  xmin = m_curves[i].x[j];
179  }
180  if (m_curves[i].y[j] < ymin) {
181  ymin = m_curves[i].y[j];
182  }
183  }
184  }
185 
186  double xd = (xmax - xmin) * 0.005;
187  double yd = (ymax - ymin) * 0.005;
188 
189  m_axisRange.xMin = xmin - xd;
190  m_axisRange.xMax = xmax + xd;
